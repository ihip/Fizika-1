\section*{Lista oznaka}

\fbox{
  \parbox[t][9ex][c]{\textwidth}{
    \begin{justify}
      Vektorske fizikalne veličine izlistane su samo kao vektori ---
      ukoliko se pojavljuju bez "strelice" radi se o iznosu ili, ako imaju indeks
      \(x\), \(y\) ili \(z\), o projekciji na odgovarajuću os (primjer: \(v\)
      je iznos trenutne brzine \(\vec{v}\), a \(v_x\) je projekcija trenutne brzine
      \(\vec{v}\) na os \(x\)).
    \end{justify}
  } %parbox
} %fbox

\begin{description}
  \item [\(A, B, C, ...\)] - oznake za točke ili tijela
  \item [\(\vec{a}, \vec{b}, \vec{c}, ...\)] - oznake za vektore
  \item [\(\vec{a}\)] - trenutno ubrzanje
  \item [\(\vec{a}_{c\!f}\)] - centrifugalno ubrzanje
  \item [\(\vec{a}_{cp}\)] - centripetalno ubrzanje
  \item [\(\vec{a}_r\)] - radijalno ubrzanje
  \item [\(\vec{a}_s\)] - srednje ubrzanje
  \item [\(\vec{a}_\tau\)] - tangencijalno ubrzanje
  \item [\(D\)] - domet kosog hica
  \item [\(d\)] - udaljenost između paralelnih osi
  \item [\(E\)] - energija
  \item [\(E_k\)] - kinetička energija
  \item [\(E_p\)] - potencijalna energija
  \item [\(E_{p,el}\)] - elastična potencijalna energija
  \item [\(E_{p,G}\)] - potencijalna energija u polju sile teže
  \item [\(E_{p,gr}\)] - gravitacijska potencijalna energija
  \item [\(E_{uk}\)] - ukupna energija
  \item [\(\vec{F}\)] - sila
  \item [\(\vec{F}_{AB}\)] - sila kojom tijelo \(A\) djeluje na tijelo \(B\)
  \item [\(\vec{F}_{cf}\)] - centrifugalna sila
  \item [\(\vec{F}_{cp}\)] - centripetalna sila
  \item [\(\vec{F}_{el}\)] - elastična sila
  \item [\(\vec{F}_I\)] - inercijska sila 
  \item [\(\vec{F}_R\)] - rezultantna sila
  \item [\(\vec{F}_{tr}\)] - sila trenja
  \item [\(\vec{F}_{tr,k}\)] - sila kinematičkog trenja
  \item [\(\vec{F}_{tr,s}\)] - sila statičkog trenja
  \item [\(\vec{F}_{tr,s,max}\)] - maksimalna sila statičkog trenja
  \item [\(\vec{F}_v\)] - vanjska sila
  \item [\(\vec{F}_\|\)] - komponenta sile koja je paralelna s pomakom
  \item [\(\vec{F}_\bot\)] - sila koja djeluje okomito na podlogu
  \item [\(\vec{G}\)] - sila teža
  \item [\(\vec{G}_\|\)] - komponenta sile teže koja je paralelna s kosinom
  \item [\(\vec{G}_\bot\)] - komponenta sile teže koja je okomita na kosinu
  \item [\(\vec{\mathcal{G}}\)] - jakost gravitacijskog polja
  \item [\(\vec{g}\)] - ubrzanja slobodnog pada, ujedno jakost polja sile teže
  \item [\(\vec{g}_0\)] - jakost gravitacijskog polja na površini Zemlje
  \item [\(H\)] - visina valjka
  \item [\(h\)] - visina
  \item [\(I\)] - moment tromosti
  \item [\(I_T\)] - moment tromosti oko osi koja prolazi kroz težište
  \item [\(\vec{i}\)] - jedinični vektor na osi \(x\)
  \item [\(\vec{j}\)] - jedinični vektor na osi \(y\)
  \item [\(K\)] - konstanta elastičnosti opruge
  \item [\(\vec{k}\)] - jedinični vektor na osi \(z\)
  \item [\(M\)] - ukupna masa sustava
  \item [\(M_Z\)] - masa Zemlje
  \item [\(m\)] - masa
  \item [\(\vec{N}\)] - moment sile
  \item [\(\hat{n}\)] - vektor normale
  \item [\(O\)] - ishodište
  \item [\(P\)] - trenutna snaga
  \item [\(P_{AB}\)] - srednja snaga
  \item [\(\vec{p}\)] - količina gibanja
  \item [\(\vec{p}_{uk}\)] - ukupna količina gibanja u zatvorenom sustavu
  \item [\(R\)] - polumjer kružnice kod kružnog gibanja
  \item [\(R_Z\)] - polumjer Zemlje
  \item [\(\vec{R}\)] - sila reakcije podloge
  \item [\(r_\bot\)] - krak sile
  \item [\(\vec{r}\)] - vektor položaja
  \item [\(\vec{r}_0\)] - početni položaj (položaj u trenutku \(t = 0\))
  \item [\(\vec{r}_{AB}\)] - vektor koji spaja točke \(A\) i \(B\) (iznos tog vektora je udaljenost između točaka)
  \item [\(\vec{r}_{cm}\)] - vektor položaja centra mase
  \item [\(\vec{r}_n\)] - položaj u kojem je opruga nerastegnuta
  \item [\(\vec{r}_{S'}\)] - vektor položaja ishodišta referentnog sustava \(S'\)
  \item [\(\Delta\vec{r}\)] - vektor pomaka, može ujedno biti i vektor deformacije
  \item [\(\Delta\vec{r}_{AB}\)] - vektor pomaka od točke \(A\) do točke \(B\)
  \item [\(S, S'\)] - oznake za referentne sustave
  \item [\(s\)] - \(s\) koordinata
  \item [\(s_A\)] - \(s\) koordinata točke \(A\)
  \item [\(T\)] - trajanje leta kod kosog hica / period rotacije Zemlje
  \item [\(\vec{T}\)] - težina / sila napetosti niti
  \item [\(t\)] - vrijeme
  \item [\(\Delta t\)] - vremenski interval
  \item [\(\vec{u}\)] - "ulazna" brzina (trenutna brzina tijela prije sudara)
  \item [\(v_1\)] - prva kozmička brzina
  \item [\(v_2\)] - druga kozmička brzina
  \item [\(\vec{v}\)] - trenutna brzina
  \item [\(\vec{v}_0\)] - početna brzina (trenutna brzina u trenutku \(t = 0\))
  \item [\(\vec{v}_s\)] - srednja brzina
  \item [\(\vec{v}_{S'}\)] - trenutna brzina referentnog sustava \(S'\)
  \item [\(W\)] - rad
  \item [\(W_{AB}\)] - rad sile na dijelu putanje između točaka \(A\) i \(B\)
  \item [\(W_G\)] - rad sile teže
  \item [\(W_{nk}\)] - rad nekonzervativnih sila
  \item [\(W_R\)] - rad rezultantne sile
  \item [\(W_{tr}\)] - rad sile trenja
  \item [\(x\)] - projekcija vektora položaja na os \(x\) (\(x\) koordinata)
  \item [\(y\)] - projekcija vektora položaja na os \(y\) (\(y\) koordinata)
  \item [\(z\)] - projekcija vektora položaja na os \(z\) (\(z\) koordinata)
  \item [\(\alpha\)] - kut nagiba kosine / trenutno kutno ubrzanje
  \item [\(\alpha_s\)] - srednje kutno ubrzanje
  \item [\(\alpha_k\)] - kut nagiba kosine pri kojem se tijelo niz kosinu giba konstantnom brzinom
  \item [\(\alpha_{max}\)] - kut nagiba kosine za koji se postiže maksimalna sila statičkog trenja
  \item [\(\gamma\)] - gravitacijska konstanta
  \item [\(\vartheta\)] - nagib kosog hica / zemljopisna širina
  \item [\(\mu_k\)] - koeficijent kinematičkog trenja
  \item [\(\mu_s\)] - koeficijent statičkog trenja
  \item [\(\rho\)] - gustoća / polumjer zakrivljenosti putanje / udaljenost od osi rotacije
  \item [\(\hat{\tau}\)] - jedinični vektor tangencijalan na putanju
  \item [\(\varphi\)] - kut zakreta kod kružnog gibanja
  \item [\(\omega\)] - trenutna kutna brzina
  \item [\(\omega_s\)] - srednja kutna brzina
\end{description}
